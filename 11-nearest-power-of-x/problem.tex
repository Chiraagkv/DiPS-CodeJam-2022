  \section*{Prompt}
  Given an integer $n$, you must find the power of $x$ that's nearest to $n$. If it's midway between the two nearest powers of $x$, round it up. 

  \subsection*{Input Format}
  \begin{itemize}
\item The first line of input contains a single integer $x$.
\item The second line of input contains a single integer $n$.
  \end{itemize}
  \subsection*{Output Format}
  The first and only line of your output must contain a single integer, the power of $x$ that's nearest to $n$.
  \subsection*{Constraints}
  \begin{itemize}
    \item $ 2 \le x \le 50 $
    \item $ 1 \le n \le 1000 $
  \end{itemize}
  \subsection*{Sample Input/Output}
  \begin{tabular}{ |l|l| } 
    \hline
    \textbf{Input} & \textbf{Output} \\
    {\lstinputlisting{./testCases/input/input00.txt}} & {\lstinputlisting{./testCases/output/output00.txt}} \\ % use {\lstinputlisting{./testCases/input/input00.txt}} & {\lstinputlisting{./testCases/output/output00.txt}}
    \hline
   \end{tabular}


  \section*{Solution}
   The two nearest powers of $x$ to $n$ are:
   \begin{align*}
    x^{\lfloor\log_nx\rfloor}\\
    x^{\lceil\log_nx\rceil}
   \end{align*}
